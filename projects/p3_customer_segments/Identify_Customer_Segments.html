# Identify missing or unknown data values and convert them to NaNs.

azdias_clean = azdias.copy()
feat_info_clean = feat_info.copy()

feat_info_clean.set_index('attribute', inplace=True)
feat_info_clean.missing_or_unknown = feat_info_clean.missing_or_unknown.str.strip('[]').str.split(',')

for col_name, col_ser in azdias_clean.iteritems():
    codes_list = feat_info_clean.loc[col_name].missing_or_unknown # codes as list of strings
    # convert list entries to appropriate datatype for that col
    if azdias_clean[col_name].dtype == "int64":
        codes_list = [int(x) for x in codes_list if x != '']
    if azdias_clean[col_name].dtype == "float64":        
        codes_list = [float(x) for x in codes_list if x != '']  
    # Replace with np.nan if an entry matches a missing code
    azdias_clean[col_name] = col_ser.apply(lambda x: x if x not in codes_list else np.nan)

def clean_data(df):
    """
    Perform feature trimming, re-encoding, and engineering for demographics
    data
    
    INPUT: Demographics DataFrame
    OUTPUT: Trimmed and cleaned demographics DataFrame
    """
    
    # Put in code here to execute all main cleaning steps:
    # convert missing value codes into NaNs, ...

    df.drop(columns=outlier_cols, inplace=True, axis=1)    

    row_threshold = 25
    row_missing_count = df.isnull().sum(axis=1)
    df = df[row_missing_count <= row_threshold]
    
    df['OST_WEST_KZ'].replace(['W','O'], [1,0], inplace=True)
    
    df.drop(columns=['CAMEO_DEU_2015'], inplace=True, axis=1)    
    categorical_cols = feat_info[feat_info.type == 'categorical'].index.values
    categorical_cols = categorical_cols[categorical_cols!='CAMEO_DEU_2015']  

    binary_categorical_with_nan_col = []
    multi_categorical_with_nan_col = []
    categorical_object_col = []

    for categorical_col in categorical_cols:
        unique_categorical_values = df[categorical_col].unique()
        if unique_categorical_values.dtype != 'object':
            has_nan = (True in np.isnan(unique_categorical_values))
            if (has_nan):
                if (len(unique_categorical_values) == 3):
                    binary_categorical_with_nan_col.append(categorical_col)
                else:
                    multi_categorical_with_nan_col.append(categorical_col)
        else:
            categorical_object_col.append(categorical_col)

    df = pd.get_dummies(df, columns = categorical_object_col, prefix = categorical_object_col)
    df = pd.get_dummies(df, columns = binary_categorical_with_nan_col, prefix = binary_categorical_with_nan_col)
    df = pd.get_dummies(df, columns = multi_categorical_with_nan_col, prefix = multi_categorical_with_nan_col)

    # select, re-encode, and engineer column values.
    mainstream_movement = [1,3,5,8,10,12,14]
    decade = {1:4, 2:4, 3:5, 4:5, 5:6, 6:6, 7:6, 8:7, 9:7, 10:8, 11:8, 12:8, 13:8, 14:9, 15:9}
    
    for val in range(1, 16):
        idx = df['PRAEGENDE_JUGENDJAHRE'] == val
        if val in mainstream_movement:
            df.loc[idx, 'IS_MAINSTREAM'] = 1
        else:
            df.loc[idx, 'IS_MAINSTREAM'] = 0
        df.loc[idx, 'DECADE'] = decade[val]
    
    for val in range(11, 56):
        idx = (df['CAMEO_INTL_2015'].str.contains(str(val),na=False))
        if df[idx].size != 0:
            df.loc[idx, 'WEALTH'] = val % 10
            df.loc[idx, 'LIFE_STAGE'] = val // 10
            
    df = df.drop('PRAEGENDE_JUGENDJAHRE', axis = 1)
    df = df.drop('CAMEO_INTL_2015', axis = 1)
    df = df.drop('LP_LEBENSPHASE_FEIN', axis = 1)
    df = df.drop('LP_LEBENSPHASE_GROB', axis = 1)

    # Return the cleaned dataframe.
    return df

Step 2.3: Interpret Principal Components¶

Now that we have our transformed principal components, it's a nice idea to check out the weight of each variable on the first few components to see if they can be interpreted in some fashion.

As a reminder, each principal component is a unit vector that points in the direction of highest variance (after accounting for the variance captured by earlier principal components). The further a weight is from zero, the more the principal component is in the direction of the corresponding feature. If two features have large weights of the same sign (both positive or both negative), then increases in one tend expect to be associated with increases in the other. To contrast, features with different signs can be expected to show a negative correlation: increases in one variable should result in a decrease in the other.
